WHERE FormStatus IN (9,2) ORDER BY  NTPersonalInfo.Post_ID,FormNo

---------------- Registrar data ---------------------------
SELECT 
       FormNo,
       FullName,
       FathersFullName,
       CAST(DOB AS DATE) DOB,	   
	   (SELECT COUNT(*) FROM dbo.PublishedWork WHERE Form_ID=RSFormPersonalDetails.Form_ID) AS NoOfPublications,
       NoOfBooksNationalPublications,
       NoOfBooksInternationalPublications,
       NoOfPatents,
       PostDoctoralFellowShip,
       UGPGTeachingExperience [UGPGTeachingExperience in Months],
       UGPGResearchExperience [UGPGResearchExperience in Months],
       ContacNo1,
       InternationalNationalLevelWards,
       StateLevelWards,
       ExamName,
       Stream,
       YearOfPassing,
       Percentage,
       BoardUniversity,
       Designation,
       NameAddressOfEmployer,
       ServiceNature,
       DateFrom,
       DateTo,
       ServiceResponsibility
FROM dbo.RSFormPersonalDetails
    JOIN dbo.RSAcademicDetails
        ON RSAcademicDetails.Form_ID = RSFormPersonalDetails.Form_ID
    JOIN dbo.RSPosts
        ON RSPosts.Post_ID = RSFormPersonalDetails.Post_ID
    LEFT JOIN dbo.RSPreviousServiceRecord
        ON RSPreviousServiceRecord.Form_ID = RSFormPersonalDetails.Form_ID
WHERE RSFormPersonalDetails.Post_ID = '57B8DD5C-9E4E-41D5-A6A8-414069CF2F6F'
AND Status<>1 
ORDER BY FormNo,RSAcademicDetails.YearOfPassing asc,ExamName,DateFrom asc,Designation

------------------------ publication count ----------------------------

SELECT COUNT(FormNo) TotalPublications,FormNo FROM dbo.RSFormPersonalDetails
LEFT JOIN dbo.PublishedWork ON PublishedWork.Form_ID = RSFormPersonalDetails.Form_ID
WHERE RSFormPersonalDetails.Post_ID = '57B8DD5C-9E4E-41D5-A6A8-414069CF2F6F'
AND Status<>1
GROUP BY FormNo


----------------------- job details --------------
SELECT 
       FormNo,
	   PostName,
	   NotificationNo,
	   NotificationDated,
       FullName,
       FathersFullName,
       CAST(DOB AS DATE) DOB,	   
       ContacNo1,
	   TxnReferenceNo,
	   TxnAmount,
	   TxnDate
FROM dbo.RSFormPersonalDetails    
		JOIN dbo.PaymentDetails ON Entity_ID=RSFormPersonalDetails.Form_ID
		JOIN dbo.RSPosts ON RSPosts.Post_ID = RSFormPersonalDetails.Post_ID
WHERE RSFormPersonalDetails.Post_ID <> '57B8DD5C-9E4E-41D5-A6A8-414069CF2F6F'
AND Status<>1  AND PostNature=2
ORDER BY RSFormPersonalDetails.Post_ID,FullName,NotificationDated
